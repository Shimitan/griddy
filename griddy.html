NON-TERMINALS

Start ::= setupPhase gamePhase <EOF>
setupPhase ::= ( setupStmt )*
setupStmt ::= functionDeclaration
| pieceDeclaration
| assignment <SEMICOLON>
| boardDeclaration <SEMICOLON>
gamePhase ::= <GAME_MAIN> <LPAREN> expression <RPAREN> ( gameStmt )*
gameStmt ::= ( functionCall <SEMICOLON> )
| outputStmt <SEMICOLON>
| assignment <SEMICOLON>
| placeStmt <SEMICOLON>
| inputStmt <SEMICOLON>
| conditionalStmt
placeStmt ::= <PLACE> identifier
position ::= <LPAREN> ( number | identifier ) <COMMA> ( number | identifier ) <RPAREN>
boardDeclaration ::= <BOARD_DECL> position
positionList ::= position ( <COMMA> position )*
pieceDeclaration ::= <PIECE_DECL> identifier ( identifier <COLON> ( number | stringLiteral | positionList | bool ) )* <END>
assignment ::= identifier <ASSIGN> expression
expression ::= ( boolExpr ( logicalOperator boolExpr )+ | boolExpr )
addExpr ::= ( multiplicativeExpr ( additiveOperator multiplicativeExpr )+ | multiplicativeExpr )
boolExpr ::= ( addExpr ( booleanOperator addExpr )+ | addExpr )
additiveOperator ::= ( <PLUS> | <MINUS> )
logicalOperator ::= ( <OR> | <AND> )
multiplicativeExpr ::= ( unary ( multiplicativeOperator unary )+ | unary )
multiplicativeOperator ::= ( <MULT> | <DIV> | <MOD> )
unary ::= <LPAREN> expression <RPAREN>
| isTileEmpty
| tileAccess
| primary
| ( <NOT> unary )
primary ::= functionCall
| ( getPiece | identifier )
| number
| bool
| stringLiteral
booleanOperator ::= ( <EQ> | <NEQ> | <LEQ> | <GEQ> | <LT> | <GT> )
identifier ::= <IDENT>
number ::= ( <NUM> | <NEG_NUM> )
bool ::= <BOOL>
stringLiteral ::= <STRING>
outputStmt ::= <OUTPUT> expression
inputStmt ::= <INPUT> expression
conditionalStmt ::= <IF> expression ( gameStmt )+ ( <END> | elseBlock )
elseBlock ::= ( <ELSE> conditionalStmt | <ELSE> ( gameStmt )+ <END> )
functionDeclaration ::= <FUNC_DECL> identifier argumentList functionBody <RETURN> expression
argumentList ::= ( <LPAREN> <RPAREN> | <LPAREN> identifier ( <COMMA> identifier )* <RPAREN> )
functionBody ::= ( gameStmt )*
functionCall ::= identifier argumentList
tileAccess ::= <TILE> position
getPiece ::= identifier <FROM> identifier
isTileEmpty ::= <IS_EMPTY> position
